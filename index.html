/** * MYlab Alumni & Team Update System - Backend v3.0 * Configuration: */ const CONFIG = { SHEET_NAME: 'Members', PHOTO_FOLDER_ID: '1UjJ1kotuHIYTUdz3G1Q2G6o0Up4AgzkT' }; /** * GET Request: Fetches the member list for the search bar */ function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(CONFIG.SHEET_NAME); if (!sheet) { return createResponse("Error: Sheet '" + CONFIG.SHEET_NAME + "' not found.", "text"); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1); const list = rows.map(row => ({ id: row[0], name: row[1], office: row[2], start: row[3], end: row[4], pos: row[5], country: row[6], currPos: row[7], topic: row[8], email: row[9], linkedin: row[10], scholar: row[11], web: row[12], other: row[13], photoUrl: row[15] || "https://via.placeholder.com/150" // Column P })); return createResponse(JSON.stringify(list), "json"); } catch (err) { return createResponse("GET Error: " + err.toString(), "text"); } } /** * POST Request: Handles the profile updates and image uploads */ function doPost(e) { console.log("Post received."); try { if (!e || !e.postData || !e.postData.contents) { throw new Error("No data received by the server."); } const payload = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(CONFIG.SHEET_NAME); const data = sheet.getDataRange().getValues(); // Find the row by ID (Column A) let rowIndex = -1; for (let i = 1; i < data.length; i++) { if (data[i][0].toString() === payload.id.toString()) { rowIndex = i + 1; break; } } if (rowIndex === -1) throw new Error("User ID " + payload.id + " not found."); // Prepare Updates for Columns B through N (Columns 2-14) // Force to String to avoid data-type mismatch errors in Sheets const updates = [ payload.name, payload.office, payload.start, payload.end, payload.pos, payload.country, payload.currPos, payload.topic, payload.email, payload.linkedin, payload.scholar, payload.web, payload.other ].map(val => val === undefined ? "" : String(val)); // Write data to the sheet sheet.getRange(rowIndex, 2, 1, updates.length).setValues([updates]); // Handle Photo Upload (Column O and P) if (payload.photoData) { const folder = DriveApp.getFolderById(CONFIG.PHOTO_FOLDER_ID); // Save new file const blob = Utilities.newBlob(Utilities.base64Decode(payload.photoData), "image/jpeg", payload.name + "_profile.jpg"); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const fileId = file.getId(); const directLink = "https://drive.google.com/uc?export=view&id=" + fileId; // Update spreadsheet with file ID and URL sheet.getRange(rowIndex, 15).setValue(fileId); sheet.getRange(rowIndex, 16).setValue(directLink); } return createResponse("Success", "text"); } catch (err) { console.error("POST Error: " + err.toString()); return createResponse("Error: " + err.toString(), "text"); } } /** * Helper to format responses correctly */ function createResponse(content, type) { const output = ContentService.createTextOutput(content); if (type === "json") { output.setMimeType(ContentService.MimeType.JSON); } else { output.setMimeType(ContentService.MimeType.TEXT); } return output; }
